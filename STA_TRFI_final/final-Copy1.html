      class fullBitstream():
          def __init__(self,fullBitstreamPath):
              self.ol = Overlay(fullBitstreamPath)
      #         self.dma=self.ol.axi_dma_0
              
          def prReconfigure(self, PR_pathName):
              decouple=GPIO(GPIO.get_gpio_pin(0),'out')
              decouple_status=GPIO(GPIO.get_gpio_pin(1),'in')
              decouple.write(1)
              partial1 = Bitstream(PR_pathName, partial=True)
              partial1.download()
              decouple.write(0)
      #         self.dma=self.ol.axi_dma_0
          
          def forward(self, matFilePath):
              mat = loadmat(matFilePath)
              rxSym = mat['Received_Symbols_FFT_Structure'].astype('complex64')
      #         staMatlabOutput = mat['STA_Structure'].astype('complex64')  # MATLAB based estOut for STA
      #         trfiMatlabOutput = mat['TRFI_Structure'].astype('complex64') # MATLAB based estOut for TRFI
              preamble  = mat['Preamble_Structure'].astype('complex64')
              act_out  = mat['True_Channels_Structure'].astype('complex64')
              preamble = np.transpose(preamble,(2,1,0))
              rxSym = np.transpose(rxSym,(2,1,0))
      #         staMatlabOutput = np.transpose(staMatlabOutput,(2,1,0))
      #         trfiMatlabOutput = np.transpose(trfiMatlabOutput,(2,1,0))
              act_out = np.transpose(act_out,(2,1,0))
               
              dma=self.ol.axi_dma_0
              send_channel=dma.sendchannel
              recv_channel=dma.recvchannel
              estOut = np.zeros((5,100,52), dtype=np.complex64)
              sendTest = preamble[0,0,:]
              input_buffer_test=allocate(shape=sendTest.shape,dtype=np.complex64)
              output_buffer_test=allocate(shape=sendTest.shape,dtype=np.complex64)
      #         err_t = 0
              err = 0
              phf = 0
              for j in range(5):
      #             print('Frame ', j)
                  sendTest = preamble[j,0,:]
                  np.copyto(input_buffer_test, sendTest)
                  input_buffer_test.flush()
                  send_channel.transfer(input_buffer_test)
                  send_channel.wait
      #             print('Preamble 1 transferred')
                  sendTest = preamble[j,1,:]
                  np.copyto(input_buffer_test, sendTest)
                  input_buffer_test.flush()
                  send_channel.transfer(input_buffer_test)
                  send_channel.wait
      #             print("Preamble Sent")
                  for i in range(100):
                      sendTest = rxSym[j,i,:]
                      np.copyto(input_buffer_test,sendTest)
                      input_buffer_test.flush()
                      send_channel.transfer(input_buffer_test)
                      send_channel.wait()
                      recv_channel.transfer(output_buffer_test)
                      recv_channel.wait()
                      output_buffer_test.invalidate()
                      estOut[j,i,:] = output_buffer_test
                      err = err+np.mean(np.abs(estOut[j,i,:] - act_out[j,i,:])**2)
                      phf = phf+np.mean(np.abs(act_out[j,i,:])**2)
                      
      #                 err_t = err_t+np.mean(np.abs(staMatlabOutput[j,i,:] - act_out[j,i,:])**2)
                 
              nmse = err/phf
      #         nmse_t = err_t/phf
      #         print('nmse = ', nmse)
              
              return nmse
